class DpiDirectoryView(APIView): def get(self, request): query = request.GET.get('search', '') if query: users = User.objects.filter(Q(username__icontains=query)) else: users = User.objects.all() return render(request, 'partials/user_cards.html', {'users': users}) def get_page(self, request): users = User.objects.all() return render(request, 'page3.html', {'users': users})
